fish <- function(v, d) {
  if (d == 0) {
    return(sum(v, digits = 999))
  } else if (d == 176) {
    print(sum(v))
  }
  fish(c(v[2:7], v[8] + v[1], v[9:10], v[2]), d - 1)
}
