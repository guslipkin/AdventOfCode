library(dplyr)
# part 1
dt <- stringr::str_split_fixed(readLines("input.txt"), pattern = "", n = 12)
dt <- data.frame(apply(dt, 2, as.integer))

getModeMin <- function(x)
  return(ifelse(mean(x) > .5, 0, 1))
getModeMax <- function(x)
  return(ifelse(mean(x) > .5, 1, 0))
